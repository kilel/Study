long long* sum(long long*a, long long*b)
{
        int p = 0, pr = 0;
        long long*c = new long long[MAX_LEN+1];
        /*for(int i = 0; i <= MAX_LEN; i++)
                c[i] = 0;*/
        c[0] = Max(a[0],b[0]);
        for(int i = a[0]+1, j = b[0]+1; i <= c[0]+1 || j <= c[0]; i++, j++)
        {
                a[i] = 0;
                b[j] = 0;
        }
        for(int s = 1; s <= c[0]; s++)
        {
                p = (a[s] + b[s] + pr)/mod;
                c[s] = (a[s] + b[s] + pr)%mod;
                pr = p;
        }
        if(pr)
        {
                c[0]++;
                c[c[0]] = pr;
        }
        return c;
}

int cmp(long long*a, long long*b)
{
        if(a[0] > b[0])
                return 1;
        if(a[0] < b[0])
                return 1;
        for(int k = a[0]; k > 0; k--)
                if(a[k] > b[k])
                        return 1;
                else if(a[k] < b[k])
                        return -1;
        return 0;
}
